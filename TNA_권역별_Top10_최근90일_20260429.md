# T&A 권역별 Top10 셀링 상품 (최근 90일, 예약건수 기준)

## 핵심 지표

| 항목 | 값 |
| --- | --- |
| 분석 기간 | 최근 90일 (배치 데이터셋 기준 KST 어제까지) |
| 권역 수 | 10 |
| 누적 결과 행 수 | 100 |
| 정렬 기준 | 예약건수 |

## Scope 요약

- 소스: `MART_SALE_D` `KIND = 1` 결제행
- 표준 T&A 6개 LV1 (`TOUR`, `TICKET`, `ACTIVITY`, `CLASS`, `CONVENIENCE`, `SNAP`)
- B2B 제외 (`SALE_FORM_CD != 'B2B'`)
- 리셀마켓 제외 (`DW_MRT_EXPERIENCES_PRODUCT_RESERVATION_CONFIGS.resell_market = TRUE` 제거)
- 패키지 제외 (`PACKAGE`, `PACKAGE_CHANNELING` 미포함)
- 2.0 ↔ 3.0 이관 dedup: `DW_MRT_EXPERIENCES_PRODUCT_MIGRATION_MAP`으로 3.0 GID 우선 통합
- AOV = `SUM(SALES_KRW_PRICE) / 예약건수`

## 사용한 쿼리
```sql
-- T&A 권역별(10개) Top10 셀링 상품 (예약건수 기준, 최근 90일)
-- scope: 표준 T&A 6 LV1, KIND=1, B2B/리셀 제외, 패키지 제외
-- canonical GID: 2.0/3.0 이관 매핑 적용 (offer_id -> 3.0 GID로 통합)

WITH base AS (
  SELECT
    s.ORDER_ID,
    s.SALES_KRW_PRICE,
    s.STANDARD_CATEGORY_LV_2_NM,
    s.PRODUCT_TITLE,
    s.CITY_CD,
    s.COUNTRY_NM,
    -- 권역 매핑
    CASE
      WHEN s.CITY_CD IN ('Nha trang','Nha Trang','CamRanh') THEN 'Nha trang'
      WHEN s.CITY_CD IN ('Sapporo') THEN 'Sapporo'
      WHEN s.CITY_CD IN ('Da Nang','Danang') THEN 'Da Nang'
      WHEN s.CITY_CD IN ('Osaka') THEN 'Osaka'
      WHEN s.CITY_CD IN ('Bohol') THEN 'Bohol'
      WHEN s.CITY_CD IN ('Bali','Jimbaran','Penida Island','Denpasar','Gili Trawangan',
                          'Seminyak','Ubud','Canggu','Nusa Dua','Kuta','Lembongan Island')
        THEN 'Bali'
      WHEN s.CITY_CD IN ('Nagoya') THEN 'Nagoya'
      WHEN s.CITY_CD IN ('Singapore') THEN 'Singapore'
      WHEN s.CITY_CD IN ('Hawaii','Honolulu','Maui','Kauai') THEN 'Hawaii'
      WHEN s.CITY_CD IN ('Shanghai') THEN 'Shanghai'
      ELSE NULL
    END AS region,
    -- 2.0/3.0 이관 dedup: 2.0 offer_id -> 3.0 GID로 묶음
    COALESCE(CAST(m.product_id AS STRING), CAST(s.GID AS STRING)) AS canonical_gid,
    s.GID AS raw_gid
  FROM `mrtdata.edw_mart.MART_SALE_D` s
  LEFT JOIN `mrtdata.edw.DW_MRT_EXPERIENCES_PRODUCT_MIGRATION_MAP` m
    ON CAST(s.GID AS STRING) = CAST(m.offer_id AS STRING)
   AND m.deleted_at IS NULL
  -- 리셀 상품 제외 (raw GID 기준으로 매칭 — resell flag는 product_id에 붙음)
  LEFT JOIN `mrtdata.edw.DW_MRT_EXPERIENCES_PRODUCT_RESERVATION_CONFIGS` rc
    ON CAST(s.GID AS STRING) = CAST(rc.product_id AS STRING)
  WHERE s.BASIS_DATE BETWEEN DATE_SUB(CURRENT_DATE('Asia/Seoul'), INTERVAL 90 DAY)
                          AND DATE_SUB(CURRENT_DATE('Asia/Seoul'), INTERVAL 1 DAY)
    AND s.KIND = 1
    AND s.STANDARD_CATEGORY_LV_1_CD IN ('TOUR','TICKET','ACTIVITY','CLASS','CONVENIENCE','SNAP')
    -- B2B 제외 (NULL은 B2C로 간주)
    AND (s.SALE_FORM_CD != 'B2B' OR s.SALE_FORM_CD IS NULL)
    -- 리셀 제외
    AND (rc.resell_market IS NOT TRUE OR rc.resell_market IS NULL)
),

filtered AS (
  SELECT *
  FROM base
  WHERE region IS NOT NULL
),

agg AS (
  SELECT
    region,
    canonical_gid,
    -- 상품명: 가장 최근/빈번한 PRODUCT_TITLE을 1개 대표값으로
    ANY_VALUE(PRODUCT_TITLE) AS sale_product_title,
    ANY_VALUE(STANDARD_CATEGORY_LV_2_NM) AS category_lv2_nm,
    COUNT(DISTINCT ORDER_ID) AS reservation_cnt,
    SUM(SALES_KRW_PRICE) AS gmv_krw,
    SAFE_DIVIDE(SUM(SALES_KRW_PRICE), COUNT(DISTINCT ORDER_ID)) AS aov_krw
  FROM filtered
  GROUP BY region, canonical_gid
),

ranked AS (
  SELECT
    region,
    canonical_gid,
    sale_product_title,
    category_lv2_nm,
    reservation_cnt,
    gmv_krw,
    aov_krw,
    ROW_NUMBER() OVER (PARTITION BY region ORDER BY reservation_cnt DESC, gmv_krw DESC) AS rn
  FROM agg
)

SELECT
  r.region                                              AS `권역`,
  r.rn                                                  AS `순위`,
  r.canonical_gid                                       AS `GID`,
  COALESCE(p.PRODUCT_NM, r.sale_product_title)          AS `상품명`,
  r.category_lv2_nm                                     AS `카테고리LV2`,
  r.reservation_cnt                                     AS `예약건수`,
  CAST(ROUND(r.gmv_krw) AS INT64)                       AS `GMV원`,
  CAST(ROUND(r.aov_krw) AS INT64)                       AS `AOV원`,
  CASE
    WHEN p.DOMAIN_NM = '3.0 PRODUCT' THEN CONCAT('https://www.myrealtrip.com/experiences/', r.canonical_gid)
    WHEN p.DOMAIN_NM = '2.0 PRODUCT' THEN CONCAT('https://www.myrealtrip.com/offers/', r.canonical_gid)
    ELSE NULL
  END                                                   AS `링크`
FROM ranked r
LEFT JOIN `mrtdata.edw_mart.MART_PRODUCT_D` p
  ON r.canonical_gid = CAST(p.GID AS STRING)
WHERE r.rn <= 10
ORDER BY
  CASE r.region
    WHEN 'Osaka' THEN 1
    WHEN 'Shanghai' THEN 2
    WHEN 'Nha trang' THEN 3
    WHEN 'Da Nang' THEN 4
    WHEN 'Bali' THEN 5
    WHEN 'Nagoya' THEN 6
    WHEN 'Hawaii' THEN 7
    WHEN 'Singapore' THEN 8
    WHEN 'Sapporo' THEN 9
    WHEN 'Bohol' THEN 10
    ELSE 99
  END,
  r.rn
```
